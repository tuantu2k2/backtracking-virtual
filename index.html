function canPutQueen(i, j):

    for each k from 0 to i - 1 do:

        if queens[k] == j or absolute(k - i) == absolute(queens[k] - j) then

            return false

    return true


function putQueen(idx):

    if not isDone then:

        for each j from 0 to n - 1 do:

            if isDone then break

            if canPutQueen(idx, j) == true then:

                queens[idx] = j

                if idx == n - 1 then:

                    isDone = true

                else:

                    call putQueen(idx + 1)             

            if j == n - 1 and not canPutQueen(idx, j) then:

                queens[idx] = 0

function canMoveKnight(x, y):

function knightTour(x, y, numC):

    if numC == col * row:

        return true

    for i from 0 to 7:

        nextX = x + knightMoves[i][0]

        nextY = y + knightMoves[i][1]

        if canMoveKnight(nextX, nextY):

            Knight[nextX][nextY] = numC

            numC++

            if knightTour(nextX, nextY, numC):

                return true

            else:

                // Backtrack if a solution is not found

                Knight[nextX][nextY] = -1

                numC--

    return false
